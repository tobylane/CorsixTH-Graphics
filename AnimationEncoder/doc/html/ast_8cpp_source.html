181  for (int i = 0; i < FN_NUMBER_ENTRIES; i++)
182  seen[i] = -1; // Haven't seen this field so far.
183 
184  for (int i = 0; i < static_cast<int>(fields.size()); i++)
185  {
186  const FieldStorage &fs = fields[i];
187 
188  // Check the assignment hasn't been done before.

190  {
191  fprintf(stderr, PROGNAME ", line %d: Field is already assigned "
192  "at line %d\n", fs.m_iLine, seen[fs.m_iNumber]);
193  exit(1);
194  }
195  seen[fs.m_iNumber] = fs.m_iLine;
196 
197  // Copy field value into a data member.
198  switch(fs.m_iNumber) {
199  case FE_TOP:

201  break;
202 
203  case FE_LEFT:
204  m_iLeft = fs.m_iValue;
205  break;
206 
207  case FE_WIDTH:
208  m_iWidth = fs.m_iValue;
209  break;
210 
211  case FE_HEIGHT:
212  m_iHeight = fs.m_iValue;
213  break;
214 
215  case FE_XOFFSET:
216  m_iXoffset = fs.m_iValue;
217  break;
218 
219  case FE_YOFFSET:
220  m_iYoffset = fs.m_iValue;
221  break;
222 
223  case FE_IMAGE:
224  m_sBaseImage = fs.m_sText;

227  case FE_RECOLOUR:

397 static void Encode32bpp(int iWidth, int iHeight, const Image32bpp &oBase, const Image8bpp *pLayer, Output *pDest, const unsigned char *pNumber)
398 {
399  const uint32 iPixCount = iWidth * iHeight;
400  uint32 iCount = 0;
401  while (iCount < iPixCount)
402  {
403  int iLength = GetDistanceToNextRecolour(iCount, iPixCount, pLayer);
404  int length2 = GetDistanceToNextTransparency(iCount, iPixCount, oBase);
405 
406  if (iLength > 63) iLength = 63;
407  if (iLength == 0) { // Recolour layer.
408  uint8 iTableNumber;
409  iLength = GetRecolourInformation(iCount, iPixCount, pLayer, &iTableNumber);
410  if (length2 < iLength) iLength = length2;
411  assert(iLength > 0);
412 
413  pDest->Uint8(64 + 128 + iLength);
414  pDest->Uint8(pNumber[iTableNumber]);
415  pDest->Uint8(GetA(oBase.Get(iCount))); // Opacity.
416  WriteTableIndex(oBase, iCount, iLength, pDest);
417  iCount += iLength;
418  continue;
419  }
420  if (length2 < iLength) iLength = length2;
421  assert(iLength > 0);
422 
423  uint8 iOpacity = GetA(oBase.Get(iCount));
424  if (iOpacity == OPAQUE) { // Fixed non-transparent 32bpp pixels (RGB).
425  pDest->Uint8(iLength);
426  WriteColour(oBase, iCount, iLength, pDest);
427  iCount += iLength;
428  continue;
429  }
430  if (iOpacity == TRANSPARENT) { // Fixed fully transparent pixels.
431  pDest->Uint8(128 + iLength);
432  iCount += iLength;
433  continue;
434  }
435  /* Partially transparent 32bpp pixels (RGB). */
436  pDest->Uint8(64 + iLength);
437  pDest->Uint8(iOpacity);
438  WriteColour(oBase, iCount, iLength, pDest);
439  iCount += iLength;
440  continue;
441  }
442 }

444 bool FrameElement::WriteSprite(Output *pOut, int *iXoffset, int *iYoffset) const
445 {
446  int iLeft = m_iLeft;
447  int iWidth = m_iWidth;
448  int iTop = m_iTop;
449  int iHeight = m_iHeight;
450  *iXoffset = m_iXoffset;
451  *iYoffset = m_iYoffset;
452 
453  Image32bpp *pBase = Load32Bpp(m_sBaseImage, m_iLine, &iLeft, &iWidth, &iTop, &iHeight, iXoffset, iYoffset);
454  if (pBase == NULL)
455  {
456  fprintf(stderr, "Warning: Skipping sprite at line %d: Image load failed.\n", m_iLine);
457  return false;
458  }
459 
460  Image8bpp *pLayer = NULL;
461  if (m_sRecolourImage != "")
462  pLayer = Load8Bpp(m_sRecolourImage, m_iLine, iLeft, iWidth, iTop, iHeight);
463 
464  int iStart = pOut->Reserve(0);
465  pOut->Uint8('S');
466  pOut->Uint8('P');
467  pOut->Uint16(iWidth);
468  pOut->Uint16(iHeight);
469  int iAddress = pOut->Reserve(4);
470  int iEnd = pOut->Reserve(0);
471  assert(iEnd - iStart == SPRITE_NON_DATA_SIZE); // Non-data size must match.
472 
473  Encode32bpp(iWidth, iHeight, *pBase, pLayer, pOut, m_aNumber);
474 
475  int iLength = pOut->Reserve(0) - (iAddress + 4); // Start counting after the length.
476  pOut->Write32(iAddress, iLength);
477 
478  delete pLayer;
479  delete pBase;
480  return true;
481 }
