55 uint32 MakeRGBA(uint8 r, uint8 g, uint8 b, uint8 a)
56 {
57  assert(sizeof(uint32) == 4); // Not really the good place, but it has to be checked somewhere!
58 
59  uint32 ret;
60  uint32 x;
61  x = r; ret = (x << PXENC_RED);
62  x = g; ret |= (x << PXENC_GREEN);
63  x = b; ret |= (x << PXENC_BLUE);
64  x = a; ret |= (x << PXENC_OPACITY);
65  return ret;
66 }

140 static void OpenFile(const std::string &sFilename, png_structp *pngPtr, png_infop *infoPtr, png_infop *endInfo, uint8 ***pRows)
141 {
142  const int PNG_SIGNATURE_SIZE = 4; // Number of bytes to read from the header for checking the given file is a PNG file.
143 
144  FILE *pFile = fopen(sFilename.c_str(), "rb");
145  if(pFile == NULL)
146  {
147  fprintf(stderr, "PNG file \"%s\" could not be opened.\n", sFilename.c_str());
148  exit(1);
149  }
150 
151  unsigned char header[PNG_SIGNATURE_SIZE];
152  if(fread(header, 1, PNG_SIGNATURE_SIZE, pFile) != PNG_SIGNATURE_SIZE)
153  {
154  fprintf(stderr, "Could not read header of \"%s\".\n", sFilename.c_str());
155  fclose(pFile);
156  exit(1);
157  }
158  bool bIsPng = !png_sig_cmp(header, 0, PNG_SIGNATURE_SIZE);
159  if(!bIsPng)
160  {
161  fprintf(stderr, "Header of \"%s\" indicates it is not a PNG file.\n", sFilename.c_str());
162  fclose(pFile);
163  exit(1);
164  }
165 
166  *pngPtr = png_create_read_struct(PNG_LIBPNG_VER_STRING, NULL, NULL, NULL);
167  if (!*pngPtr)
168  {
169  fprintf(stderr, "Could not initialize PNG data.\n");
170  fclose(pFile);
171  exit(1);
172  }
173  *infoPtr = png_create_info_struct(*pngPtr);
174  if(!*infoPtr)
175  {
176  fprintf(stderr, "Could not initialize PNG info data.\n");
177  png_destroy_read_struct(pngPtr, (png_infopp)NULL, (png_infopp)NULL);
178  fclose(pFile);
179  exit(1);
180  }
181 
182  *endInfo = png_create_info_struct(*pngPtr);
183  if(!*endInfo)
184  {
185  fprintf(stderr, "Could not initialize PNG end data.\n");
186  png_destroy_read_struct(pngPtr, infoPtr, (png_infopp)NULL);
187  fclose(pFile);
188  exit(1);
189  }
190 
191  /* Setup callback in case of errors. */
192  if(setjmp(png_jmpbuf(*pngPtr))) {
193  fprintf(stderr, "Error detected while reading PNG file.\n");
194  png_destroy_read_struct(pngPtr, infoPtr, endInfo);
195  fclose(pFile);
196  exit(1);
197  }
198 
199  /* Initialize for file reading. */
200  png_init_io(*pngPtr, pFile);
201  png_set_sig_bytes(*pngPtr, PNG_SIGNATURE_SIZE);
202 
203  png_read_png(*pngPtr, *infoPtr, PNG_TRANSFORM_IDENTITY, NULL);
204  *pRows = png_get_rows(*pngPtr, *infoPtr);
205  fclose(pFile);
206 }

218 static void PerformCropping(uint8 **pRows, int *left_edge, int *top_edge, int *width, int *height, int *xoffset, int *yoffset)
219 {
220  int xpoint = *left_edge - *xoffset;
221  int ypoint = *top_edge - *yoffset;
222  int left = *left_edge;
223  int top = *top_edge;
224  int right = left + *width - 1;
225  int bottom = top + *height - 1;
226 
227  while (left <= right)
228  {
229  bool ok = true;
230  for (int y = top; y <= bottom; y++)
231  {
232  uint8 *pPixel = pRows[y] + left;
233  if (pPixel[CH_OPACITY] != TRANSPARENT)
234  {
235  ok = false;
236  break;
237  }
238  }
239  if (!ok)
240  break;
241 
242  left++;
243  }
244  while (left <= right)
245  {
246  bool ok = true;
247  for (int y = top; y <= bottom; y++)
248  {
249  uint8 *pPixel = pRows[y] + right;
250  if (pPixel[CH_OPACITY] != TRANSPARENT)
251  {
252  ok = false;
253  break;
254  }
255  }
256  if (!ok)
257  break;
258 
259  right--;
260  }
261 
262  while (top <= bottom)
263  {
264  bool ok = true;
265  for (int x = left; x <= right; x++)
266  {
267  uint8 *pPixel = pRows[top] + x;
268  if (pPixel[CH_OPACITY] != TRANSPARENT)
269  {
270  ok = false;
271  break;
272  }
273  }
274  if (!ok)
275  break;
276 
277  top++;
278  }
279  while (top <= bottom)
280  {
281  bool ok = true;
282  for (int x = left; x <= right; x++)
283  {
284  uint8 *pPixel = pRows[bottom] + x;
285  if (pPixel[CH_OPACITY] != TRANSPARENT)
286  {
287  ok = false;
288  break;
289  }
290  }
291  if (!ok)
292  break;
293 
294  bottom--;
295  }
296 
297  *left_edge = left;
298  *top_edge = top;
299  *width = right - left + 1;
300  *height = bottom - top + 1;
301  *xoffset = left - xpoint;
302  *yoffset = top - ypoint;
303 }

305 Image32bpp *Load32Bpp(const std::string &sFilename, int line, int *left, int *width, int *top, int *height, int *xoffset, int *yoffset)
306 {
307  png_structp pngPtr;
308  png_infop infoPtr;
309  png_infop endInfo;
310  uint8 **pRows;
311  OpenFile(sFilename, &pngPtr, &infoPtr, &endInfo, &pRows);
312 
313  int iWidth = png_get_image_width(pngPtr, infoPtr);
314  int iHeight = png_get_image_height(pngPtr, infoPtr);
315  int iBitDepth = png_get_bit_depth(pngPtr, infoPtr);
316  int iColorType = png_get_color_type(pngPtr, infoPtr);
317 
318  /* Initialize sprite width and height if not set, clamping at 0. */
319  if (*width < 0)
320  *width = iWidth - *left;
321  if (*width < 0)
322  *width = 0;
323 
324  if (*height < 0)
325  *height = iHeight - *top;
326  if (*height < 0)
327  *height = 0;
328 
329  /* Test whether the sprite fits in the image. */
330  if (iWidth < *left + *width)
331  {
332  fprintf(stderr, "Sprite at line %d: Sprite is not wide enough, require %d columns (%d + %d) while only %d columns are available.\n", line, *left + *width, *left, *width, iWidth);
333  exit(1);
334  }
335  if (iHeight < *top + *height)
336  {
337  fprintf(stderr, "Sprite at line %d: Sprite is not high enough, require %d rows (%d + %d) while only %d rows are available.\n", line, *top + *height, *top, *height, iHeight);
338  exit(1);
339  }
340 
341  if (iBitDepth != 8)
342  {
343  fprintf(stderr, "Sprite at line %d: \"%s\" is not an 32bpp file (channels are not 8 bit wide)\n", line, sFilename.c_str());
344  png_destroy_read_struct(&pngPtr, &infoPtr, &endInfo);
345  exit(1);
346  }
347  if (iColorType != PNG_COLOR_TYPE_RGB_ALPHA)
348  {
349  fprintf(stderr, "Sprite at line %d: \"%s\" is not an RGBA file\n", line, sFilename.c_str());
350  png_destroy_read_struct(&pngPtr, &infoPtr, &endInfo);
351  exit(1);
352  }
353 
354  PerformCropping(pRows, left, top, width, height, xoffset, yoffset);
355  if (*width == 0 || *height == 0)
356  {
357  fprintf(stderr, "Sprite at line %d: \"%s\" is empty\n", line, sFilename.c_str());
358  png_destroy_read_struct(&pngPtr, &infoPtr, &endInfo);
359  exit(1);
360  }
361 
362  Image32bpp *img = new Image32bpp(*width, *height);
363  uint32 *pData = img->pData;
364  for (int i = 0; i < *height; i++)
365  {
366  uint8 *pRow = pRows[*top + i] + (*left * RGBA_CHANNELS_PER_PIXEL);
367  for (int j = 0; j < *width; j++)
368  {
369  *pData++ = MakeRGBA(pRow[CH_RED], pRow[CH_GREEN], pRow[CH_BLUE], pRow[CH_OPACITY]);
370  pRow += RGBA_CHANNELS_PER_PIXEL;
371  }
372  }
373 
374  png_destroy_read_struct(&pngPtr, &infoPtr, &endInfo);
375  return img;
376 }

416  uint8 *pData = img->pData;

420  for (int j = 0; j < width; j++)
421  {
422  int x = left + j;
423  uint8 v = pRows[y][x];

425  pData++;
426  }
427  }
