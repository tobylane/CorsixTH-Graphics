364 static int EncodeSprite(const FrameElement &fe, int *iXoffset, int *iYoffset, Output *output)
365 {
366  // Encode the sprite.
367  Output out;
368  if (!fe.WriteSprite(&out, iXoffset, iYoffset))
369  {
370  fprintf(stderr, "Warning: Sprite \"%s\" cannot be created, using sprite 0\n", fe.m_sBaseImage.c_str());
371  return 0;
372  }
373 
374  EncodedSprite oEncSprite;
375  oEncSprite.TakeData(out.GetData(), out.GetSize());
376 
377  // Find sprite in written sprites.
378  std::map<EncodedSprite, int>::iterator iter;
379  iter = g_mapSprites.find(oEncSprite);
380  if (iter != g_mapSprites.end())
381  return (*iter).second;
382 
383  // Write sprite block.
384  g_iTotalSpriteSize += oEncSprite.m_iSize - SPRITE_NON_DATA_SIZE; // Subtract header length.
385  for (int idx = 0; idx < oEncSprite.m_iSize; idx++)
386  output->Uint8(oEncSprite.m_pData[idx]);
387 
388  // Store sprite for future re-use.
389  std::pair<EncodedSprite, int> p(oEncSprite, g_mapSprites.size());
390  iter = g_mapSprites.insert(p).first;
391  return (*iter).second;
392 }

400 static int EncodeFrames(const Animation &an, Output *output)
401 {
402  int first_frame = g_iNumberWrittenFrames;
403 
404  for (FrameConstIterator fri = an.m_vFrames.begin(); fri != an.m_vFrames.end(); fri++)
405  {
406  const AnimationFrame &fr = *fri;
407 
408  // Encode the elements of the frame, storing the results temporarily.
409  std::vector<SpriteElement> elements;
410  for (ElementConstIterator ei = fr.m_vElements.begin(); ei != fr.m_vElements.end(); ei++)
411  {
412  SpriteElement se;
413  se.m_iSprite = EncodeSprite(*ei, &se.m_iXoffset, &se.m_iYoffset, output);
414  if ((*ei).m_oDisplay.m_iKey < 0)
415  {
416  se.m_iLayerclass = 0;
417  se.m_iLayerId = 0;
418  }
419  else
420  {
421  se.m_iLayerclass = (*ei).m_oDisplay.m_iKey;
422  se.m_iLayerId = (*ei).m_oDisplay.m_iValue;
423  }
424  se.m_iFlags = 0;
425  if ((*ei).m_bVertFlip) se.m_iFlags |= 0x1;
426  if ((*ei).m_bHorFlip) se.m_iFlags |= 0x2;
427  if ((*ei).m_iAlpha == 50) se.m_iFlags |= 0x4;
428  if ((*ei).m_iAlpha == 75) se.m_iFlags |= 0x8;
429 
430  elements.push_back(se);
431  }
432 
433  // Encode the frame.
434  output->Uint8('F');
435  output->Uint8('R');
436  output->Uint16((fr.m_iSound < 0) ? 0 : fr.m_iSound);
437  output->Uint16(elements.size());
438  for (std::vector<SpriteElement>::iterator iter = elements.begin(); iter != elements.end(); iter++)
439  {
440  output->Uint32((*iter).m_iSprite);
441  output->Uint16((*iter).m_iXoffset);
442  output->Uint16((*iter).m_iYoffset);
443  output->Uint8((*iter).m_iLayerclass);
444  output->Uint8((*iter).m_iLayerId);
445  output->Uint16((*iter).m_iFlags);
446 
447  g_iTotalElements++;
448  }
449 
450  g_iNumberWrittenFrames++;
451  }
452  return first_frame;
453 }

460 static void EncodeAnimationGroup(const AnimationGroup &ag, Output *output)
461 {
462  unsigned int first_frames[4];
463 
464  // Find an animation.
465  const Animation *an = NULL; // Find an animation.
466  for (int idx = 0; idx < 4; idx++)
467  {
468  if (ag.m_aAnims[idx] == NULL)
469  continue;
470 
471  an = ag.m_aAnims[idx];
472  break;
473  }
474  if (an == NULL)
475  {
476  fprintf(stderr, "Animation group without animation, skipping it\n");
477  return;
478  }
479 
480  // Encode all frames
481  for (int idx = 0; idx < 4; idx++)
482  {
483  if (ag.m_aAnims[idx] == NULL)
484  {
485  first_frames[idx] = 0xFFFFFFFF;
486  }
487  else
488  {
489  first_frames[idx] = EncodeFrames(*ag.m_aAnims[idx], output);
490  }
491  }
492 
493  // Output grouped animation.
494  output->Uint8('C');
495  output->Uint8('A');
496  output->Uint16(an->m_iTileSize);
497  output->Uint32(an->m_vFrames.size());
498  output->String(an->m_sName);
499  for (int idx = 0; idx < 4; idx++)
500  output->Uint32(first_frames[idx]);
501 }

503 void Encode(const char *outFname)
504 {
505  g_iNumberWrittenFrames = 0;
506  g_iTotalElements = 0;
507  g_mapSprites.clear();
508  g_iTotalSpriteSize = 0;
509 
510  Output output;
511 
512  output.Uint8('C');
513  output.Uint8('T');
514  output.Uint8('H');
515  output.Uint8('G');
516  output.Uint16(512+1);
517  output.Uint32(g_mapAnimGroups.size());
518  int iAddrTotalFrames = output.Reserve(4);
519  int iAddrTotalElements = output.Reserve(4);
520  int iAddrTotalSprites = output.Reserve(4);
521  int iAddrSumSpriteSize = output.Reserve(4);
522 
523  for (GroupIterator grp = g_mapAnimGroups.begin(); grp != g_mapAnimGroups.end(); grp++)
524  {
525  EncodeAnimationGroup((*grp).second, &output);
526  }
527 
528  output.Write32(iAddrTotalFrames, g_iNumberWrittenFrames);
529  output.Write32(iAddrTotalElements, g_iTotalElements);
530  output.Write32(iAddrTotalSprites, g_mapSprites.size());
531  output.Write32(iAddrSumSpriteSize, g_iTotalSpriteSize);
532 
533  output.Write(outFname);
534 }
